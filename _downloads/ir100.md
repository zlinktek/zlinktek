(async function () {
  const dataUrl = "{{ '/assets/data/ir100_firmware.json' | relative_url }}";
  const input = document.getElementById("fw-sn");
  const list  = document.getElementById("fw-list");
  const stat  = document.getElementById("fw-status");

  function esc(s){ return String(s).replace(/[&<>"']/g, m => ({'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;'}[m])); }

  function render(index, q) {
    list.innerHTML = "";
    const keys = Object.keys(index).sort();
    const filtered = q ? keys.filter(k => k.includes(q)) : keys;

    stat.textContent = `共 ${keys.length} 台记录；当前显示 ${filtered.length} 条。`;

    if (filtered.length === 0) {
      list.innerHTML = "<li>未找到匹配编号。</li>";
      return;
    }

    for (const sn of filtered) {
      const url = index[sn];
      const li = document.createElement("li");
      li.innerHTML = `<a href="${esc(url)}" target="_blank" rel="noopener">ZL-IR100-${esc(sn)}</a>`;
      list.appendChild(li);
    }
  }

  try {
    stat.textContent = "正在加载固件索引...";
    const res = await fetch(dataUrl, { cache: "no-store" });
    if (!res.ok) throw new Error("固件索引加载失败");
    const index = await res.json();

    render(index, "");

    input.addEventListener("input", () => {
      const q = (input.value || "").trim();
      render(index, q);
    });
  } catch (e) {
    stat.textContent = "加载失败：" + (e && e.message ? e.message : e);
  }
})();
</script>
